JavaScript Data Binding using EventListener

 

JavaScript Data Binding using onInput

 


  //html
    <section>
      <h3>JavaScript Data Binding using EventListener</h3>
      <input id="inputbox" type="text" 
                           placeholder="Start typing ..." />
      <div id="outputdiv"> </div>
    </section>

    <section>
      <h3>JavaScript Data Binding using onInput</h3>
      <input id="inputbox2" type="text" 
                            placeholder="Start typing here" />
      <div id="outputdiv2"> </div>
    </section>

  <script type="text/javascript" src="app.js"></script> 

  \\ js

(function() {

  var outputdiv = document.getElementById('outputdiv'),
    outputdiv2 = document.getElementById('outputdiv2'),
    inputbox = document.getElementById('inputbox'),
    inputbox2 = document.getElementById('inputbox2');

  function change(a,b) {
    a.innerHTML = b.value; 
  }

  if (inputbox) {
    inputbox.addEventListener('input', 
      change.bind(null, outputdiv, inputbox), false);
  } 

  if (inputbox2) {
    inputbox2.oninput = function () {
      change(outputdiv2, inputbox2);
    }
  } 

}());
